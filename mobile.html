Why WebPlus?

HTML, CSS and JavaScript Support

WebPlus supports HTML, CSS and JavaScript compilation

Easy to Learn

WebPlus syntax is similar to JavaScript and Python and hence is easy to learn and understand

Built-in Auto-Builder and Live-Server

WebPlus comes with built-in Auto-Builder and customisable Live-Server

Installing the CLI

Terminal

pip install webplus

Creating New Project

Terminal

webplus create <project name>

Example Project

main.py

from modules.document import Document
from modules.element import Element
from modules.script import Script
from modules.stylesheet import Stylesheet

def main():
doc = Document("Home", "index.html")

doc.body[ "class" ] = "bg-dark"

doc.importStylesheet("./bootstrap-5.0.2-dist/css/bootstrap.min.css")

div = Element("div")
div.id = "div_1"
div.className = "container"
doc.append(div)

p = Element("p")
p.id = "p_1"
p.className = "p"
p.innerText = "Hello there"
div.append(p)

doc.ready()

CSS

main.py

from modules.document import Document
from modules.stylesheet import Stylesheet

def main():
doc = Document("Home", "index.html")

# import external stylesheet
doc.importStylesheet("./bootstrap-5.0.2-dist/css/bootstrap.min.css")

# create stylesheet
style = Stylesheet( "style.css")
style.add(".tab", {
"background": "#212529"
"color": "#ffffff"
})

doc.ready()
